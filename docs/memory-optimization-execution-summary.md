## 💡 优化建议

### 短期改进（明天执行）

#### 1. 启用正常模式（最优先）

修改 `.env`:
```env
MEMORY_BACKFILL_LOW_RESOURCE_MODE=false
MEMORY_BACKFILL_RSS_RECYCLE_MB=512
MEMORY_BACKFILL_RSS_GROWTH_MB=128
MEMORY_EMBEDDING_BACKFILL_BATCH_SIZE=64
MEMORY_EMBEDDING_BACKFILL_MAX_PER_RUN=500
```

**预期提升**: 每轮 500 节点 → **速度提升 5 倍**

#### 2. 自动化回填脚本

创建 `scripts/auto-backfill-loop.js`:
```javascript
#!/usr/bin/env node
const { execSync } = require('child_process');

async function autoBackfillLoop() {
  const TARGET_COVERAGE = 0.60;
  let round = 1;
  
  while (true) {
    console.log(`\n=== Auto Backfill Round ${round} ===`);
    
    // 执行回填
    execSync('node scripts/backfill-memory-v3-embeddings.js --resume --source memory --limit 500 --sync-after', {
      stdio: 'inherit'
    });
    
    // 检查覆盖率
    const output = execSync('npm run diag:memory -- diagnose --skip-probe', { encoding: 'utf-8' });
    const coverage = JSON.parse(output).summary.coverage.memory.readyRatio;
    
    console.log(`Current coverage: ${(coverage * 100).toFixed(1)}%`);
    
    if (coverage >= TARGET_COVERAGE) {
      console.log(`✅ Target coverage ${TARGET_COVERAGE * 100}% reached!`);
      break;
    }
    
    round++;
    console.log('Waiting 2 hours before next round...');
    await new Promise(resolve => setTimeout(resolve, 2 * 60 * 60 * 1000));
  }
}

autoBackfillLoop().catch(console.error);
```

**用法**:
```bash
node scripts/auto-backfill-loop.js
```

#### 3. 处理失败节点

```bash
# 查看失败详情
npm run diag:memory -- backfill --dry-run | jq '.failureBreakdown'

# 强制重试
node scripts/backfill-memory-v3-embeddings.js \
  --resume \
  --source memory \
  --retry-failed \
  --force \
  --limit 100
```
