**预期提升**: 速度 ×5，每轮 500 节点

#### 2. 处理失败节点

```bash
# 查看失败详情
npm run diag:memory -- backfill --dry-run | jq '.failureBreakdown'

# 强制重试
node scripts/backfill-memory-v3-embeddings.js \
  --resume --source memory --retry-failed --force --limit 100
```

**目标**: 失败节点从 81 → 0

#### 3. 自动化回填脚本

创建 `scripts/auto-backfill-to-target.js`:
```javascript
#!/usr/bin/env node
const { execSync } = require('child_process');

const TARGET_COVERAGE = 0.60;
const BATCH_SIZE = 500;
const SLEEP_MINUTES = 120;  // 2小时

async function runBackfillLoop() {
  let round = 1;
  
  while (true) {
    console.log(`\n=== Auto Backfill Round ${round} ===`);
    console.log(`Target: ${TARGET_COVERAGE * 100}%, Batch: ${BATCH_SIZE}`);
    
    try {
      // 执行回填
      execSync(`node scripts/backfill-memory-v3-embeddings.js --resume --source memory --limit ${BATCH_SIZE} --sync-after`, {
        stdio: 'inherit'
      });
      
      // 检查覆盖率
      const output = execSync('npm run diag:memory -- diagnose --skip-probe', { encoding: 'utf-8' });
      const match = output.match(/"readyRatio":\s*([\d.]+)/);
      const coverage = match ? parseFloat(match[1]) : 0;
      
      console.log(`\n✅ Round ${round} completed`);
      console.log(`Current coverage: ${(coverage * 100).toFixed(1)}%`);
      
      if (coverage >= TARGET_COVERAGE) {
        console.log(`\n🎉 Target coverage ${TARGET_COVERAGE * 100}% reached!`);
        break;
      }
      
      round++;
      const sleepMs = SLEEP_MINUTES * 60 * 1000;
      console.log(`\nSleeping ${SLEEP_MINUTES} minutes before next round...`);
      await new Promise(resolve => setTimeout(resolve, sleepMs));
      
    } catch (error) {
      console.error(`\n❌ Round ${round} failed:`, error.message);
      console.log('Retrying in 10 minutes...');
      await new Promise(resolve => setTimeout(resolve, 10 * 60 * 1000));
    }
  }
}

runBackfillLoop().catch(console.error);
```

**用法**:
```bash
# 前台运行（推荐先测试）
node scripts/auto-backfill-to-target.js

# 后台运行（Linux/Mac）
nohup node scripts/auto-backfill-to-target.js > backfill.log 2>&1 &

# 后台运行（Windows PowerShell）
Start-Process node -ArgumentList "scripts/auto-backfill-to-target.js" -WindowStyle Hidden
```
